    <h3>카테고리</h3>

    <input id="categoryInput" placeholder="카테고리 추가" />

    <button onclick="addCategory()">추가</button>

    <div id="categories"></div>

  </div>


  <div class="box">

    <h3>항목</h3>

    <input id="itemInput" placeholder="항목 추가 (한글)" />

    <input id="itemEnInput" placeholder="프롬프트 문구 (영문)" />

    <button onclick="addItem()">추가</button>

    <div id="items"></div>

  </div>


  <div class="box">

    <h3>프롬프트</h3>

    <textarea id="promptOutput" rows="12"></textarea>

    <button onclick="copyPrompt()">복사</button>

    <button onclick="clearPrompt()">초기화</button>

  </div>

<script>

let data = JSON.parse(localStorage.getItem("promptData")) || {};

let currentCategory = null;


function save() {

  localStorage.setItem("promptData", JSON.stringify(data));

}


function addCategory() {

  const name = categoryInput.value.trim();

  if (!name) return;

  data[name] = [];

  currentCategory = name;

  categoryInput.value = "";

  save();

  renderCategories();

}


function renderCategories() {

  categories.innerHTML = "";

  Object.keys(data).forEach(cat => {

    const div = document.createElement("div");

    div.className = "item";

    div.innerText = cat;

    div.onclick = () => {

      currentCategory = cat;

      renderItems();

    };

    categories.appendChild(div);

  });

}


function addItem() {

  if (!currentCategory) return alert("카테고리를 선택하세요");

  const ko = itemInput.value.trim();

  const en = itemEnInput.value.trim();

  if (!ko || !en) return;

  data[currentCategory].push({ ko, en });

  itemInput.value = "";

  itemEnInput.value = "";

  save();

  renderItems();

}


function renderItems() {

  items.innerHTML = "";

  data[currentCategory].forEach(obj => {

    const div = document.createElement("div");

    div.className = "item";

    div.innerText = obj.ko;

    div.onclick = () => {

      promptOutput.value += obj.en + ", ";

    };

    items.appendChild(div);

  });

}


function copyPrompt() {

  promptOutput.select();

  document.execCommand("copy");

  alert("복사됨!");

}


function clearPrompt() {

  promptOutput.value = "";

}


renderCategories();


if ('serviceWorker' in navigator) {

  navigator.serviceWorker.register('service-worker.js');

}

</script>
